//从尾到头打印链表节点(常规方法) 时间复杂度(O(n)) 空间复杂度(O(1))
public void versePrint(){

    if(head == null){
        System.out.println("链表为空");
        return;
    }

    ListNode p = head.next;
    ListNode q = null;
    head.next = null;

    while (p != null){
        q = p.next;
        p.next = head;
        head = p;
        p = q;
    }
    print();
}

//栈实现 时间复杂度:O(n) 空间复杂度:O(n)
public void versePrint1(){
    Stack<Integer> stack = new Stack<Integer>();

    ListNode p = head;
    if(p == null){
        System.out.println("链表为空");
        return;
    }

    while (p != null){
        stack.push(p.data);
        p = p.next;
    }

    while (!stack.empty()){
        System.out.print(stack.pop()+"-->");
    }
}

//递归实现
public void versePrint2(ListNode p){
    if(p == null){
        return;
    }
    versePrint2(p.next);
    System.out.print(p.data+"-->");
}

 //链表中的倒数第 K 个结点
public ListNode FindKthToTail(ListNode head,int k){

    ListNode fast = head;
    ListNode slow = head;

    if(k <= 0){
        throw new RuntimeException("请输入自然数k！");
    }

    if(head == null) {
        throw new RuntimeException("链表不能为空！");
    }

    for(int i = 0;i < k;i++){
        if(fast != null){//先走k步
            fast = fast.next;
        }else {
            throw new RuntimeException("k值大于了链表长度");
        }
    }

    while (fast != null){//pre == null 时p指针指向倒数第k个结点
        slow = slow.next;
        fast = fast.next;
    }
    return slow;
}

合并两个有序数组

发表于 2018-07-22 | 分类于 数组 | 阅读次数:
  • 两个有序数组A1和A2，把A2的所有数字插入到A1？

核心思想：从数组末尾开始比较，指针依次往前挪。

public class MergeSortedArray {

//时间复杂度(O(n)) 空间复杂度(O(n)) 通过归并实现
public static int[] merge(int[] a,int[] b){

    if(a==null || b==null){
        System.out.println("输入两个数组");
    }

    if(a.length == 0)
        return b;

    if(b.length == 0)
        return a;

    int len1 = a.length;
    int len2 = b.length;

    int[] c = new int[len1+len2];
    int i = 0;
    int j = 0;

    while(i < len1 && j < len2){//归并的思想
        if(a[i] < b[j]){
            c[i+j] = a[i];
            i++;
        }else{
            c[i+j] = b[j];
            j++;
        }
    }

    while (i < len1){
        c[i+j+1] = a[i];
        i++;
    }

    while(j < len2){
        c[i+j] = b[j];
        j++;
    }

    return c;
}

//时间复杂度(O(n)) 空间复杂度(O(1)) 直接在第一个数组上操作
public static void merge1(int[] a,int[] b){

    if(a==null || b==null){
        System.out.println("输入两个数组");
    }

    int len1 = a.length;
    int len2 = b.length;

    int i = len1 - len2 -1;
    int j = len2 - 1;

    while(i >= 0 && j >= 0){//从数组最后面的元素开始向前比较
        if(a[i] > b[j]){
            a[i+j+1] = a[i];
            i--;
        }else{
            a[i+j+1] = b[j];
            j--;
        }
    }

    while(j >= 0){
        a[i+j+1] = b[j];
        j--;
    }

}

public static void main(String[] args){

    int[] a ={1,4,9};
    int[] b = {2,3,10,16};
    int[] c = new int[a.length+b.length];

    c = MergeSortedArray.merge(a,b);

    //打印数组元素
    System.out.println(Arrays.toString(c));
}
}

替换字符串字符

发表于 2018-07-20 | 分类于 字符串 | 阅读次数:
  • 将字符串中的’ ‘替换成”20%”？

核心思想：

1.从最后一字符开始从前遍历，最后一个字符是’\0’。

2.将字符串变为数组时，要在字符串数组后面添加’\0’。

public class ReplaceChar {

//时间复杂度：O(n) 空间复杂度：O(n)
public static void replace(char[] a,int len){

    //len = 1 时就一个字符'\0'
    if(a == null || len<2){
        System.out.println("输入长度大于0的字符串");
        return;
    }

    int oldLen = 0;
    int count = 0;
    int newLen;
    for(int i = 0;a[i] !='\0';i++){
        if(a[i] == ' ')
            count++;
        oldLen++;
    }
    newLen = oldLen + 2*count;
    //判断输入与字符数组的不匹配的字符串长度
    if(newLen > len)
        return;

    int j = newLen - 1;
    for(int i = oldLen-1;i >= 0;i--) {
        if (a[i] == ' ') {
            a[j--] = '0';
            a[j--] = '2';
            a[j--] = '%';
        } else {
            //注意 j 越界的情况
            a[j--] = a[i];
        }
    }
}

2.右上角 > key 去列 否则 去行；

public class Solution {
public boolean Find(int target, int [][] array) {
    if(array == null){
        return false;
    }
    int row = array.length;
    int col = array[0].length;
    int i = 0;
    int j = array[0].length - 1;
    while(i < row && j >= 0){
        int rightUpCorner = array[i][j];
        if(rightUpCorner < target){
            i++;
        }else if(rightUpCorner > target){
            j--;
        }else{
            return true;
        }
    }
    return false;
    }
}

数组中的重复数字

发表于 2018-07-18 | 分类于 数组 | 阅读次数:

长度为n的数组里面的所有数字都在0~n-1的范围内，找出数组中的重复数字？

核心思想：

1.数组下标 = 数组元素

2.目标交换：num[i] = m

swap: m <---> num[m] 直到 num[i] == i

public class OptDuplication {

//哈希表的方法解决，时间复杂度：O(n) 空间复杂度：O(n)
boolean findDup1(int[] nums,int len,int[] duplication){

    if(nums == null || len <= 0){
        System.out.println("数组必须存在且不能为空!");
        return  false;
    }

    for(int i = 0;i<len;i++){
        if(nums[i]<0 || nums[i]>len-1) {
            System.out.println("数组元素应在0 ~ len-1");
            return false;
        }
    }

    //哈希表
    int[] count = new int[len];
    for(int i = 0;i<len;i++){
        if(count[nums[i]] == 1){
            duplication[0] = nums[i];
            return true;
        }

        count[nums[i]]++;
    }

    return false;

}

boolean findDup(int[] nums,int len,int[] duplication){

    if(nums == null || len <= 0){
        System.out.println("数组必须存在且不能为空!");
        return  false;
    }

    for(int i = 0;i<len;i++){
        if(nums[i]<0 || nums[i]>len-1) {
            System.out.println("数组元素应在0 ~ len-1");
            return false;
        }
    }

    int temp;
    for(int i = 0;i<len;i++){
        while (i != nums[i]){
            //nums[i]=m 说明 m 所在的下标是 i 和 m 所以重复了
            if(nums[i] == nums[nums[i]]){
                duplication[0]=nums[i];
                return true;
            }else {
                //交换num[i] 和 num[num[i]]
                temp = nums[i];
                nums[i] = nums[nums[i]];
                nums[temp] = temp;
            }
        }
    }

    System.out.println("没有重复数字");
    return false;
}

长度为n+1的数组里面的所有数字都在1~n的范围内，所以必定存在重复的数字，找出数组中的重复数字？

核心思想：

1.二分法的运用，加入了统计数字个数的方法

2.循环结束条件：start == end

public class DuplicationVersion {

//时间复杂度:O(nlogn) 空间复杂度:O(1)
boolean findDup(int[] nums,int len,int[] duplication){

    if(nums == null || len <= 0){
        System.out.println("数组必须存在且不能为空!");
        return  false;
    }

    for(int i = 0;i<len;i++){
        if(nums[i]<=0 || nums[i]>len-1) {
            System.out.println("数组元素应在1 ~ "+ (len-1)+ "之间");
            return false;
        }
    }

    int start = 1;
    int end = len-1;
    int middle;
    while(start <=  end){

        middle = (start + end)/2;
        int number = calNum(nums,len,start,middle);

        //结束条件
        if(start == end){
            if(number > 1){
                duplication[0] = start;
                return true;
            }else{
                break;
            }
        }

        if(middle - start + 1 < number){
            end = middle;
        }else {
            start = middle + 1;
        }
    }

    System.out.println("没有重复数字");
    return false;
}

private int calNum(int[] nums, int len, int start, int end) {
    int count = 0;
    for(int i = 0;i<len;i++){
        if(nums[i] >= start && nums[i] <= end){
            count++;
        }
    }
    return count;
}
 }

  • 阻塞队列

    阻塞队列会对当前线程产生阻塞，比如一个线程从一个空的阻塞队列中取元素，此时线程会被阻塞直到阻塞队列中有了元素。
当队列中有元素后，被阻塞的线程会自动被唤醒（不需要我们编写代码去唤醒）。这样提供了极大的方便性。

  • wait()、notify()和notifyAll()

    1.如果调用某个对象的 wait()/notify() 方法，当前线程必须拥有这个对象的monitor（即锁），因此调用 wait()/notify()
方法必须在同步块或者同步方法中进行（synchronized块或者synchronized方法)。

2.调用某个对象的wait()方法，相当于让当前线程交出此对象的monitor，然后进入等待状态，等待后续再次获得此对象的锁。（而sleep不会释放该对象的锁）

3.notifyAll() 唤醒所有正在等待该对象的monitor的线程。

4.一个线程被唤醒不代表立即获取了对象的monitor，只有等调用完notify()或者notifyAll()并退出synchronized块，释放对象锁后，其余线程才可获得锁执行。

  • Condition

它用来替代传统的Object的wait()、notify()实现线程间的协作,更加安全和高效。

1.Condition依赖于Lock接口，生成一个Condition的基本代码是lock.newCondition()。

2.调用Condition的await()和signal()方法，都必须在lock保护之内，就是说必须在lock.lock()和lock.unlock之间才可以使用。

3.
    Conditon中的await()对应Object的wait()；
    Condition中的signal()对应Object的notify()；
    Condition中的signalAll()对应Object的notifyAll()。

private int dowait(boolean timed, long nanos)
    throws InterruptedException, BrokenBarrierException,
           TimeoutException {
    final ReentrantLock lock = this.lock;
    lock.lock();
    try {
        final Generation g = generation;

        if (g.broken)
            throw new BrokenBarrierException();

        if (Thread.interrupted()) {
            breakBarrier();
            throw new InterruptedException();
        }

        int index = --count;
        if (index == 0) {  // tripped
            boolean ranAction = false;
            try {
                final Runnable command = barrierCommand;
                if (command != null)
                    command.run();//调用构造函数中的Runnable实现
                ranAction = true;
                nextGeneration();//唤醒所有阻塞的线程
                return 0;
            } finally {
                if (!ranAction)
                    breakBarrier();
            }
        }


        for (;;) {
            try {
                if (!timed)
                    trip.await();//线程还没到齐就一直阻塞
                else if (nanos > 0L)
                    nanos = trip.awaitNanos(nanos);
            } catch (InterruptedException ie) {
                if (g == generation && ! g.broken) {
                    breakBarrier();
                    throw ie;
                } else {
                    // We're about to finish waiting even if we had not
                    // been interrupted, so this interrupt is deemed to
                    // "belong" to subsequent execution.
                    Thread.currentThread().interrupt();
                }
            }

            if (g.broken)
                throw new BrokenBarrierException();

            if (g != generation)
                return index;

            if (timed && nanos <= 0L) {
                breakBarrier();
                throw new TimeoutException();
            }
        }
    } finally {
        lock.unlock();
    }
}

单例模式

单例模式是如何保证单例的？

即怎么保证内存中只有一个对象：

1.外界不能随意创建对象，所有构造函数必须私有化；

2.类本身要有一个对象。即类里面要封装好一个对象，下次通过公共方法调用时都是返回的这个对象；

3.通过公共的方法提供给别人。因为外界不能创建对象，所以通过给方法加static关键字，通过类名访问类方法来获取对象；

  • 单例模式指？

    只生成一个对象，避免重复创建对象。

  • instance为什么要加static关键字？

    因为单例模式中构造器被私有化，产生实例的唯一途径就是通过 static 方法，在类加载的时候创建对象，这样就保证对象的唯一性。。。而static方法中只能访问静态变量,所以必须声明成static.

  • 单例模式中构造函数为什么私有化？

    1.构造函数私有化后，那么只能通过这个类的内部函数才能构造这个类的对象。

    2.将生成类对象的方法用static修饰的原因是：因为静态方法不需要依赖实例化的对象。

  • 单例模式中volatile变量的作用？

    首先分析 instance = new Singleton() 的本质：

    1.为对象分配内存空间

    2.对象初始化

    3.将对象的引用指向对应的内存地址。

此时线程B访问的是没有经过初始化的对象，会报错，所以使用volatile的根源在于：禁止A2和A3的重排序。

  • 单例模式中为什么要使用双重检查索(DCL)？

    因为在多线程的环境下，所有线程在调用getSingleton()方法时都必须在synchronized这里排队，而加锁是一个非常耗时的操作，这导致单例模式的效率非常低。我们只需要在实例还没有创建之前添加加锁操作。

  • 线程安全的延迟初始化？

    1.基于volatile

    2.基于类初始化

    对于2举个例子：

  • 静态内部类方法创建单例模式的好处？

    按需创建实例（饿汉式模式：在类加载时就完成了初始化），提高了内存的利用效率。

    这种方式的特点是：Singleton类被装载了，instance不一定被初始化，只有显示通过调用getInstance方法时，才会显示装载SingletonHolder类，从而实例化instance。

  • 栈内存和堆内存的比较？

**堆内存存的内容：**数组元素、类对象、类中的非静态成员变量（实例变量）、包装类数据(Integer等)、

**栈内存内容：**

1.方法中声明的变量(因为方法执行时，被分配的内存就在栈中,即方法栈)：分基本数据类型变量和引用变量，前者的变量名和值都存在方法栈中，后者声明的变量存放在栈中，所指的对象放在堆中。

2.对象的引用

**方法区：**线程共享的内存区域。类里面的静态成员和静态方法、

**常量池：**方法区的一块内存。

**静态域：**方法区的一块内存。存放类中的静态成员变量

  • 局部变量保存在栈内存中，随方法的消失而消失;成员变量存储在堆中的对象里，由垃圾回收器负责回收。

  • 包装类数据Integer、String等等存在于堆中。
